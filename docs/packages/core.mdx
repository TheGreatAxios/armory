# @armory/base

Core protocol types, EIP-712 signing, encoding, network configs, and token registry.

## Installation

```bash
bun add @armory/base
```

## Key Exports

```typescript
import {
  // Types
  type PaymentPayload,
  type PaymentPayloadV1,
  type PaymentPayloadV2,
  type SettlementResponse,

  // Encoding/Decoding
  encodePaymentV2,
  decodePaymentV2,
  decodePayment,

  // EIP-712
  createEIP712Domain,
  createTransferWithAuthorization,

  // Networks
  getNetworkConfig,
  getNetworkByChainId,

  // Token Registry
  registerToken,
  getCustomToken,
  type CustomToken,
} from '@armory/base';
```

## Network Configuration

```typescript
import { getNetworkConfig } from '@armory/base';

const base = getNetworkConfig(8453);
// { name: 'base', chainId: 8453, caip2Id: 'eip155:8453', ... }
```

## Custom Tokens

```typescript
import { registerToken, type CustomToken } from '@armory/base';

const myToken: CustomToken = {
  symbol: 'MYTOKEN',
  name: 'My Custom Token',
  version: '1',
  contractAddress: '0x...',
  chainId: 8453,
  decimals: 18,
};

registerToken(myToken);
```
